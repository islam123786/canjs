push

Add items to the end of the list.

var l = new can.List([]);

l.bind('change', function(
    ev,        // the change event
    attr,      // the attr that was changed, for multiple items, "*" is used
    how,       // "add"
    newVals,   // an array of new values pushed
    oldVals,   // undefined
    where      // the location where these items where added
    ) {

})

l.push('0','1','2');
